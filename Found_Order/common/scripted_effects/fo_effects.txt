make_new_character_female_if_feminist = { # scope is new character
  if = {
    limit = {
      OR = {
        has_religion_feature = religion_equal
        has_religion_feature = religion_feature_west_african
        ROOT = { has_law = status_of_women_4 }
      }
    }
    random = {
      chance = 20
      modifier = {
        factor = 1.4
        ROOT = { is_female = yes }
      }
      # Audax Validator "." Ignore_ALL
      set_gender = female
    }
  }
  if = {
    limit = { has_religion_feature = religion_matriarchal }
    # Audax Validator "." Ignore_ALL
    set_gender = female
  }
}
